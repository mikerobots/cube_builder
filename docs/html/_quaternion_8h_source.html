141 // Roll (x-axis rotation)
142 float sinr_cosp = 2 * (w * x + y * z);
143 float cosr_cosp = 1 - 2 * (x * x + y * y);
144 angles.x = std::atan2(sinr_cosp, cosr_cosp);
145
146 // Pitch (y-axis rotation)
147 float sinp = 2 * (w * y - z * x);
148 if (std::abs(sinp) >= 1)
149 angles.y = std::copysign(M_PI / 2, sinp); // use 90 degrees if out of range

151 angles.y = std::asin(sinp);

153 // Yaw (z-axis rotation)
154 float siny_cosp = 2 * (w * z + x * y);
155 float cosy_cosp = 1 - 2 * (y * y + z * z);
156 angles.z = std::atan2(siny_cosp, cosy_cosp);

190 static Quaternion lookRotation(const Vector3f& forward, const Vector3f& up = Vector3f::UnitY()) {
191 Vector3f f = forward.normalized();
192 Vector3f u = up.normalized();
193 Vector3f r = f.cross(u).normalized();
194 u = r.cross(f);
195
196 float trace = r.x + u.y + f.z;
197 if (trace > 0.0f) {
198 float s = std::sqrt(trace + 1.0f) * 2.0f;
199 return Quaternion(
200 (u.z - f.y) / s,
201 (f.x - r.z) / s,
202 (r.y - u.x) / s,
203 0.25f * s
204 );
205 } else if (r.x > u.y && r.x > f.z) {
206 float s = std::sqrt(1.0f + r.x - u.y - f.z) * 2.0f;
207 return Quaternion(
208 0.25f * s,
209 (r.y + u.x) / s,
210 (f.x + r.z) / s,
211 (u.z - f.y) / s
212 );
213 } else if (u.y > f.z) {
214 float s = std::sqrt(1.0f + u.y - r.x - f.z) * 2.0f;
215 return Quaternion(
216 (r.y + u.x) / s,
217 0.25f * s,
218 (u.z + f.y) / s,
219 (f.x - r.z) / s
220 );
221 } else {
222 float s = std::sqrt(1.0f + f.z - r.x - u.y) * 2.0f;
223 return Quaternion(
224 (f.x + r.z) / s,
225 (u.z + f.y) / s,

232 static Quaternion slerp(const Quaternion& a, const Quaternion& b, float t) {
233 float dot = a.dot(b);
234
235 // If the dot product is negative, slerp won't take the shorter path.
236 Quaternion b_adjusted = (dot < 0.0f) ? Quaternion(-b.x, -b.y, -b.z, -b.w) : b;
237 dot = std::abs(dot);
238
239 // If the inputs are too close for comfort, linearly interpolate
240 if (dot > 0.9995f) {
241 Quaternion result = a + (b_adjusted - a) * t;
242 return result.normalized();
243 }
244
245 float theta_0 = std::acos(dot);
246 float sin_theta_0 = std::sin(theta_0);
247 float theta = theta_0 * t;
248 float sin_theta = std::sin(theta);
249
250 float s0 = std::cos(theta) - dot * sin_theta / sin_theta_0;
251 float s1 = sin_theta / sin_theta_0;
252
253 return (a * s0) + (b_adjusted * s1);
254 }
