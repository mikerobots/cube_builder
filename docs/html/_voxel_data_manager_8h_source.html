316 // Performance metrics
317 struct PerformanceMetrics {
318 size_t totalVoxels;
319 size_t totalMemoryUsage;
320 float memoryEfficiency;
321 std::array<size_t, static_cast<int>(VoxelResolution::COUNT)> voxelsByResolution;
322 std::array<size_t, static_cast<int>(VoxelResolution::COUNT)> memoryByResolution;
323 };
324
325 PerformanceMetrics getPerformanceMetrics() const {
326 std::lock_guard<std::mutex> lock(m_mutex);
327
328 PerformanceMetrics metrics = {};
329
330 for (int i = 0; i < static_cast<int>(VoxelResolution::COUNT); ++i) {
331 if (m_grids[i]) {
332 metrics.voxelsByResolution[i] = m_grids[i]->getVoxelCount();
333 metrics.memoryByResolution[i] = m_grids[i]->getMemoryUsage();
334 metrics.totalVoxels += metrics.voxelsByResolution[i];
335 metrics.totalMemoryUsage += metrics.memoryByResolution[i];
336 }
337 }
338
339 // Calculate overall memory efficiency
340 if (metrics.totalMemoryUsage > 0) {
341 size_t theoreticalMinMemory = metrics.totalVoxels * sizeof(bool);
342 metrics.memoryEfficiency = static_cast<float>(theoreticalMinMemory) /
343 static_cast<float>(metrics.totalMemoryUsage);
344 } else {
345 metrics.memoryEfficiency = 1.0f;
346 }
347
348 return metrics;
349 }
