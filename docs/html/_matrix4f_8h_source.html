299 static Matrix4f perspective(float fovRadians, float aspect, float nearPlane, float farPlane) {
300 float tanHalfFov = std::tan(fovRadians * 0.5f);
301 float range = farPlane - nearPlane;
302
303 Matrix4f result;
304 result.m[0] = 1.0f / (aspect * tanHalfFov);
305 result.m[5] = 1.0f / tanHalfFov;
306 result.m[10] = -(farPlane + nearPlane) / range;
307 result.m[11] = -2.0f * farPlane * nearPlane / range;
308 result.m[14] = -1.0f;
309 result.m[15] = 0.0f;
310 return result;
311 }

313 static Matrix4f orthographic(float left, float right, float bottom, float top, float nearPlane, float farPlane) {
314 Matrix4f result;
315 result.m[0] = 2.0f / (right - left);
316 result.m[3] = -(right + left) / (right - left);
317 result.m[5] = 2.0f / (top - bottom);

320 result.m[11] = -(farPlane + nearPlane) / (farPlane - nearPlane);

324 static Matrix4f lookAt(const Vector3f& eye, const Vector3f& center, const Vector3f& up) {
325 Vector3f f = (center - eye).normalized();
326 Vector3f s = f.cross(up).normalized();
327 Vector3f u = s.cross(f);
328
329 Matrix4f result;
330 result.m[0] = s.x;
331 result.m[1] = s.y;
332 result.m[2] = s.z;
333 result.m[3] = -s.dot(eye);
334
335 result.m[4] = u.x;
336 result.m[5] = u.y;
337 result.m[6] = u.z;
338 result.m[7] = -u.dot(eye);
339
340 result.m[8] = -f.x;
341 result.m[9] = -f.y;
342 result.m[10] = -f.z;
343 result.m[11] = f.dot(eye);
344
345 result.m[12] = 0.0f;
346 result.m[13] = 0.0f;
347 result.m[14] = 0.0f;
348 result.m[15] = 1.0f;
