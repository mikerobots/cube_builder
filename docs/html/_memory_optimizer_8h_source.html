43 void handleEvent(const Events::MemoryPressureEvent& event) override {
44 if (!m_enabled) return;
45
46 float pressureRatio = static_cast<float>(event.currentUsage) / event.maxUsage;
47
48 if (pressureRatio > 0.95f) {
49 performCleanup(CleanupPriority::Critical);
50 } else if (pressureRatio > 0.90f) {
51 performCleanup(CleanupPriority::High);
52 } else if (pressureRatio > 0.80f) {
53 performCleanup(CleanupPriority::Medium);
54 }
55 }

58 m_cleanupCallbacks.emplace_back(callback, priority, name);
59
60 std::sort(m_cleanupCallbacks.begin(), m_cleanupCallbacks.end(),
61 [](const CleanupCallback& a, const CleanupCallback& b) {

63 });
64 }

66 size_t performCleanup(CleanupPriority minPriority = CleanupPriority::Low) {
67 size_t totalFreed = 0;
68
69 for (const auto& callback : m_cleanupCallbacks) {
70 if (static_cast<int>(callback.getPriority()) >= static_cast<int>(minPriority)) {
71 size_t freed = callback.execute();
72 totalFreed += freed;
73
74 if (freed > 0) {
75 // Log cleanup action
76 }
