244 void updateSmoothing(float deltaTime) {
245 bool changed = false;
246 float lerpFactor = 1.0f - std::pow(1.0f - m_smoothFactor, deltaTime * 60.0f); // 60fps reference
247
248 // Smooth distance
249 if (std::abs(m_distance - m_targetDistance) > 0.001f) {
250 m_distance = Math::lerp(m_distance, m_targetDistance, lerpFactor);
251 changed = true;
252 }
253
254 // Smooth angles
255 if (std::abs(m_yaw - m_targetYaw) > 0.01f) {
256 m_yaw = Math::lerp(m_yaw, m_targetYaw, lerpFactor);
257 changed = true;
258 }
259
260 if (std::abs(m_pitch - m_targetPitch) > 0.01f) {
261 m_pitch = Math::lerp(m_pitch, m_targetPitch, lerpFactor);
262 changed = true;
263 }
264
265 // Smooth target position
266 Math::Vector3f targetDelta = m_targetTarget - m_target;
267 if (targetDelta.length() > 0.001f) {
268 Math::Vector3f newTarget = m_target + targetDelta * lerpFactor;
