33 void recordAllocation(void* ptr, size_t size, const char* category = "Unknown") {
34 if (!ptr || size == 0) return;
35
36 std::lock_guard<std::mutex> lock(m_mutex);
37
38 m_allocations.emplace(ptr, AllocationInfo(ptr, size, category));
39
40 m_totalAllocated.fetch_add(size, std::memory_order_relaxed);
41 size_t currentUsage = m_currentUsage.fetch_add(size, std::memory_order_relaxed) + size;
42
43 if (currentUsage > m_peakUsage.load(std::memory_order_relaxed)) {
44 m_peakUsage.store(currentUsage, std::memory_order_relaxed);
45 }
46
47 m_categoryUsage[category] += size;
48
49 checkMemoryPressure(currentUsage);
50 }

52 void recordDeallocation(void* ptr) {
53 if (!ptr) return;
54
55 std::lock_guard<std::mutex> lock(m_mutex);
56
57 auto it = m_allocations.find(ptr);
58 if (it != m_allocations.end()) {
59 size_t size = it->second.size;
60 const std::string& category = it->second.category;
61
62 m_totalDeallocated.fetch_add(size, std::memory_order_relaxed);
63 m_currentUsage.fetch_sub(size, std::memory_order_relaxed);
64
65 m_categoryUsage[category] -= size;
66 if (m_categoryUsage[category] == 0) {
67 m_categoryUsage.erase(category);
68 }
69
70 m_allocations.erase(it);
71 }
72 }
