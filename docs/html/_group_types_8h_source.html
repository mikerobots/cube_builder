20// VoxelId structure for group membership
21struct VoxelId {
22 Math::Vector3i position;
23 VoxelData::VoxelResolution resolution;
24
25 VoxelId() = default;
26 VoxelId(const Math::Vector3i& pos, VoxelData::VoxelResolution res)
27 : position(pos), resolution(res) {}
28
29 bool operator==(const VoxelId& other) const {
30 return position == other.position && resolution == other.resolution;
31 }
32
33 bool operator!=(const VoxelId& other) const {
34 return !(*this == other);
35 }
36
37 size_t hash() const {
38 size_t h1 = std::hash<int>{}(position.x);
39 size_t h2 = std::hash<int>{}(position.y);
40 size_t h3 = std::hash<int>{}(position.z);
41 size_t h4 = std::hash<int>{}(static_cast<int>(resolution));
42
43 // Combine hashes
44 return h1 ^ (h2 << 1) ^ (h3 << 2) ^ (h4 << 3);
45 }
46};

112struct GroupTransform {

119
120 bool isIdentity() const {
