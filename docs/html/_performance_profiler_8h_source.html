43 void endSection(const std::string& name) {
44 std::lock_guard<std::mutex> lock(m_mutex);
45
46 if (m_sectionStack.empty()) {
47 return; // No matching begin
48 }
49
50 auto top = m_sectionStack.top();
51 if (top.name != name) {
52 return; // Mismatched section names
53 }
54
55 m_sectionStack.pop();
56
57 auto now = std::chrono::high_resolution_clock::now();
58 auto duration = std::chrono::duration<double, std::milli>(now - top.startTime).count();
59
60 auto& data = m_sections[name];
61 data.name = name;
62 data.totalTime += duration;
63 data.callCount++;
64 data.averageTime = data.totalTime / data.callCount;
65 data.minTime = std::min(data.minTime, duration);
66 data.maxTime = std::max(data.maxTime, duration);
67 }
