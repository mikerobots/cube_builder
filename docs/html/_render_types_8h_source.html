131// Vertex layout description
132struct VertexLayout {
133 struct Attribute {
134 VertexAttribute attribute;
135 int components; // 1-4
136 VertexAttributeType type;
137 bool normalized;
138 size_t offset;
139
140 Attribute(VertexAttribute attr, int comp, VertexAttributeType t, bool norm = false, size_t off = 0)
141 : attribute(attr), components(comp), type(t), normalized(norm), offset(off) {}
142 };
143
144 std::vector<Attribute> attributes;
145 size_t stride = 0;
146
147 void addAttribute(VertexAttribute attr, int components, VertexAttributeType type, bool normalized = false) {
148 attributes.emplace_back(attr, components, type, normalized, stride);
149 size_t typeSize = 4; // Assume 4 bytes for now (float/int)
150 if (type == VertexAttributeType::Byte || type == VertexAttributeType::UByte) {
151 typeSize = 1;
152 }
153 stride += components * typeSize;
154 }
155
156 void clear() {
157 attributes.clear();
158 stride = 0;
159 }
160};
