# Minimal CMake file for render test
cmake_minimum_required(VERSION 3.10)

# Add the test executable
add_executable(test_render_core test_render_core.cpp)

# Link required libraries
target_link_libraries(test_render_core
    VoxelEditor_Rendering
    VoxelEditor_Camera
    VoxelEditor_Math
    VoxelEditor_Events
    VoxelEditor_Logging
    VoxelEditor_VoxelData
    glad
    glfw
    ${OPENGL_LIBRARIES}
)

# Include directories
target_include_directories(test_render_core PRIVATE
    ${CMAKE_SOURCE_DIR}
    ${CMAKE_SOURCE_DIR}/foundation
    ${CMAKE_SOURCE_DIR}/core
    ${CMAKE_SOURCE_DIR}/external/glad/include
)
